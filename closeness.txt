Closeness = (string1,string2) ->
  arr = string1.split('/')
  arr2 = string2.split('/')
  len = Math.min(arr.length,arr2.length)
  closeness = 0
  for index in [0..len] 
    if arr[index] is arr2[index]
      closeness++

  return closeness
